local isGlobalDragging = false

local function makeDraggable(gui, saveKey)
    local dragging, dragStart, startPos = false, nil, nil
    local dragThreshold = 5
    gui.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true; dragStart = input.Position; startPos = gui.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false; task.wait(0.05); isGlobalDragging = false
                    config[saveKey] = {gui.Position.X.Scale, gui.Position.X.Offset, gui.Position.Y.Scale, gui.Position.Y.Offset}
                    save()
                end
            end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            if delta.Magnitude > dragThreshold then isGlobalDragging = true end
            if isGlobalDragging then
                local targetPos = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
                gui.Position = gui.Position:Lerp(targetPos, 0.5)
            end
        end
    end)
end

-- FINALIZED CHANNEL-BASED AUTO LEAVE (StealHistory Filter)
task.spawn(function()
    local targetName = player.Name
    local channelFolder = ReplicatedStorage:WaitForChild("Packages"):WaitForChild("Synchronizer"):WaitForChild("Channel")
    
    local function connectRemote(remote)
        if remote:IsA("RemoteEvent") and (remote.Name == targetName or remote.Name == "Events") then
            remote.OnClientEvent:Connect(function(data)
                if config.autoLeave and typeof(data) == "table" then
                    for _, action in pairs(data) do
                        local path = action[1]
                        local pathStr = typeof(path) == "table" and table.concat(path, ".") or tostring(path)
                        if pathStr:find("StealHistory") then
                            player:Kick("wow u stole a brainrot")
                            break
                        end
                    end
                end
            end)
        end
    end

    for _, v in pairs(channelFolder:GetChildren()) do connectRemote(v) end
    channelFolder.ChildAdded:Connect(connectRemote)
end)
